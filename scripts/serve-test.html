#!/usr/bin/env node /** * Simple HTTP server to serve the browser test HTML file * This allows testing the SDK in a real browser environment */ const http = require('http'); const fs = require('fs'); const path = require('path'); const PORT = 8000; const TEST_FILE = path.join(__dirname, '../tests/browser-test.html'); const DIST_DIR = path.join(__dirname, '../dist'); const NODE_MODULES_DIR = path.join(__dirname, '../node_modules'); const MIME_TYPES = { '.html': 'text/html', '.js': 'application/javascript', '.mjs': 'application/javascript', '.json': 'application/json', '.css': 'text/css', '.wasm': 'application/wasm', '.map': 'application/json', }; const server = http.createServer((req, res) => { let filePath; // Handle root request if (req.url === '/' || req.url === '/tests/browser-test.html') { filePath = TEST_FILE; } else { // Allow access to dist and node_modules for browser build if (req.url.startsWith('/dist/')) { filePath = path.join(DIST_DIR, req.url.substring('/dist/'.length)); } else if (req.url.startsWith('/node_modules/')) { filePath = path.join(NODE_MODULES_DIR, req.url.substring('/node_modules/'.length)); } else { filePath = path.join(__dirname, '..', req.url); } } // Security: prevent directory traversal filePath = path.normalize(filePath); const rootDir = path.normalize(path.join(__dirname, '..')); if (!filePath.startsWith(rootDir)) { res.writeHead(403); res.end('Forbidden'); return; } // Check if file exists if (!fs.existsSync(filePath)) { res.writeHead(404); res.end('Not Found'); return; } const ext = path.extname(filePath); const contentType = MIME_TYPES[ext] || 'application/octet-stream'; fs.readFile(filePath, (err, data) => { if (err) { res.writeHead(500); res.end('Error loading file'); return; } res.writeHead(200, { 'Content-Type': contentType }); res.end(data); }); }); server.listen(PORT, () => { const url = `http://localhost:${PORT}/tests/browser-test.html`; console.log('\nðŸš€ Browser test server started!'); console.log(`\nðŸ“– Open in your browser:`); console.log(` ${url}\n`); console.log('Press Ctrl+C to stop the server\n'); });
